#!/bin/bash
sudo apt update
sudo apt install apache2 \
                 ghostscript \
                 libapache2-mod-php \
                 mysql-server \
                 php \
                 php-bcmath \
                 php-curl \
                 php-imagick \
                 php-intl \
                 php-json \
                 php-mbstring \
                 php-mysql \
                 php-xml \
                 php-zip
sudo mkdir -p /srv/www
sudo chown www-data: /srv/www
curl https://wordpress.org/latest.tar.gz | sudo -u www-data tar zx -C /srv/www
wget https://github.com/leonie04/Modul-346/blob/main/Configs/wordpress.conf
sudo mv ./wordpress.conf /etc/apache2/sites-available/
sudo a2ensite wordpress
sudo a2enmod rewrite
sudo a2dissite 000-default
sudo service apache2 reload
wget https://github.com/leonie04/Modul-346/blob/main/Configs/MySQL_Setup.sql
sudo mysql -u root < MySQL_Setup.sql
sudo service mysql start
wget https://github.com/leonie04/Modul-346/blob/main/Configs/wp-config.php
sudo -u www-data mv ./wp-config.php /srv/www/wordpress/
